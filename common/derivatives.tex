% total derivatives
\newcommand{\td}[2]         { \frac{d #1} {d #2} }
\newcommand{\tdt}[1]        { \frac{d #1} {dt} }
\newcommand{\tdx}[1]        { \frac{d #1} {dx} }

\newcommand{\std}[2]        { \frac{d^[2] #1} {d #2^[2]} }
\newcommand{\stdt}[1]       { \frac{d #1^[2]} {dt^[2]} }
\newcommand{\stdx}[1]       { \frac{d #1^[2]} {dx^[2]} }

% partial derivatives
\newcommand{\pd}[2]         { \frac{\partial #1} {\partial #2} }
\newcommand{\pdt}[1]        { \frac{\partial #1} {\partial t} }
\newcommand{\pdx}[1]        { \frac{\partial #1} {\partial x} }

\newcommand{\spd}[2]        { \frac{\partial^[2] #1} {\partial #2^[2]} }
\newcommand{\spdt}[1]       { \frac{\partial #1^[2]} {\partial  t^[2]} }
\newcommand{\spdx}[1]       { \frac{\partial #1^[2]} {\partial  x^[2]} }
