\newcommand{\quarterdisk}[1]    { 
    \begin{tikzpicture}
    	\draw (0,0) arc (0:90:#1);
    	\draw (0,0) -- (-#1,0);
    	\draw (-#1,0) -- (-#1,#1);
    \end{tikzpicture}
}

\newcommand{\qradial}[1]    { 
    \begin{tikzpicture}
    	\draw[thick] (0,0) arc (0:90:#1);
    	\draw (0,0) -- (-#1,0);
    	\draw (-#1,0) -- (-#1,#1);
    \end{tikzpicture}
}
